def catalan(n):
    if n <= 0:
        return 1  # Base case
    else:
        result = 0
        for i in range(n):
            result += catalan(i) * catalan(n - i - 1)
        return result

# Example usage
if __name__ == '__main__':
    n = 5  # Change this number to generate a different Catalan number
    print(f'The {n}th Catalan number is: {catalan(n)}')
