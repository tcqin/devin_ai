def fibonacci(n):
    if n <= 0:
        return "Please enter a positive integer"
    elif n == 1:
        return 0
    elif n == 2:
        return 1
    else:
        a, b = 0, 1
        for _ in range(2, n):
            a, b = b, a + b
        return b

# Example usage
if __name__ == '__main__':
    n = 10  # Change this number to generate a different Fibonacci number
    print(f'The {n}th Fibonacci number is: {fibonacci(n)}')
